# Production Cron Jobs Configuration
# Use this for cloud servers (AWS EC2, DigitalOcean, etc.)

# Environment setup
SHELL=/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/sbin:/bin:/usr/sbin:/usr/bin

# Load environment variables
. /opt/sa-insight-hub/.env.production

# Realtime scrapers (forex + energy) - Every 30 minutes
*/30 * * * * /opt/sa-insight-hub/cron_production.sh realtime "forex energy" "data: realtime update — forex + energy [skip ci]" "data/forex.json data/energy.json data/manifest.json"

# Weekly scrapers (water + finance + property) - Every Monday at 06:00 UTC (08:00 SAST)
0 6 * * 1 /opt/sa-insight-hub/cron_production.sh weekly "water finance property" "data: weekly update — water + finance + property [skip ci]" "data/water.json data/finance.json data/property.json data/manifest.json"

# Quarterly scrapers (all 10 topics) - 1st of January, April, July, October at 04:00 UTC
0 4 1 1,4,7,10 * /opt/sa-insight-hub/cron_production.sh quarterly "crime forex water finance energy employment health education property fraud" "data: quarterly full refresh — all 10 topics [skip ci]" "data/*.json"

# Health check - Every 5 minutes to ensure system is running
*/5 * * * * /opt/sa-insight-hub/health_check.sh

# Log cleanup - Delete logs older than 30 days (runs daily at 02:00)
0 2 * * * find /opt/sa-insight-hub/logs -name "*.log" -mtime +30 -delete

# Backup data directory - Daily at 03:00 UTC
0 3 * * * tar -czf /opt/backups/sa-insight-hub-$(date +\%Y-\%m-\%d).tar.gz -C /opt/sa-insight-hub data/ logs/
